def dailyGei
  geiFinal = 0
  nodo = @head

  while(nodo != nil)
    geiFinal += nodo.value.gei
    nodo = nodo.next
  end
  return geiFinal.round(1)
end

def extractHead
  if @head != nil
    auxNodo = @head
    
    @head = head.next
    
    if (@head == nil)
      @tail = nil
    else
      @head.prev = nil
    end

    auxNodo.prev = nil
    return auxNodo
  end
end

def insertHead(value)
  auxNodo = Node.new(value) # Nodo a insertar
  if (@head != nil)
    @head.prev = auxNodo
    auxNodo.next = @head
    @head = auxNodo
    @head.prev = nil
  else
    @head = auxNodo
  end
end
